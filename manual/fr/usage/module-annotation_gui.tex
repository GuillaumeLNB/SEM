\begin{itemize}
    \item[] \textbf{description}
        \begin{itemize}
            \item[] Interface graphique pour annoter manuellement des fichiers.
        \end{itemize}
    \item[] \textbf{arguments}
        \begin{itemize}
            \item[] aucun argument.
        \end{itemize}
    \item[] \textbf{options}
        \begin{itemize}
            \item[] --help ou -h : switch
                \begin{itemize}
                    \item[] affiche l'aide
                \end{itemize}
            \item[] --log ou -l : string
                \begin{itemize}
                    \item[] définit le niveau de log : info, warn ou critical (défaut : critical).
                \end{itemize}
        \end{itemize}
\end{itemize}
